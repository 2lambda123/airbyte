216 def should_retry(self, response: requests.Response) -> bool: 

217 """ 

218 Override to set different conditions for backoff based on the response from the server. 

219 

220 By default, back off on the following HTTP response statuses: 

221 - 429 (Too Many Requests) indicating rate limiting 

222 - 500s to handle transient server errors 

223 

224 Unexpected but transient exceptions (connection timeout, DNS resolution failed, etc..) are retried by default. 

225 """ 

226 return response.status_code == 429 or 500 <= response.status_code < 600

239 def _create_prepared_request( 

240 self, path: str, headers: Mapping = None, params: Mapping = None, json: Any = None, data: Any = None 

241 ) -> requests.PreparedRequest: 

242 args = {"method": self.http_method, "url": self.url_base + path, "headers": headers, "params": params} 

243 if self.http_method.upper() in BODY_REQUEST_METHODS: 

244 if json and data: 

245 raise RequestBodyException( 

246 "At the same time only one of the 'request_body_data' and 'request_body_json' functions can return data" 

247 ) 

248 elif json: 

249 args["json"] = json 

250 elif data: 

251 args["data"] = data 

252 

253 return self._session.prepare_request(requests.Request(**args))

255 def _send(self, request: requests.PreparedRequest, request_kwargs: Mapping[str, Any]) -> requests.Response: 

256 """ 

257 Wraps sending the request in rate limit and error handlers. 

258 Please note that error handling for HTTP status codes will be ignored if raise_on_http_errors is set to False 

259 

260 This method handles two types of exceptions: 

261 1. Expected transient exceptions e.g: 429 status code. 

262 2. Unexpected transient exceptions e.g: timeout. 

263 

264 To trigger a backoff, we raise an exception that is handled by the backoff decorator. If an exception is not handled by the decorator will 

265 fail the sync. 

266 

267 For expected transient exceptions, backoff time is determined by the type of exception raised: 

268 1. CustomBackoffException uses the user-provided backoff value 

269 2. DefaultBackoffException falls back on the decorator's default behavior e.g: exponential backoff 

270 

271 Unexpected transient exceptions use the default backoff parameters. 

272 Unexpected persistent exceptions are not handled and will cause the sync to fail. 

273 """ 

274 response: requests.Response = self._session.send(request, **request_kwargs) 

275 

276 if self.should_retry(response): 

277 custom_backoff_time = self.backoff_time(response) 

278 if custom_backoff_time: 

279 raise UserDefinedBackoffException(backoff=custom_backoff_time, request=request, response=response) 

280 else: 

281 raise DefaultBackoffException(request=request, response=response) 

282 elif self.raise_on_http_errors: 

283 # Raise any HTTP exceptions that happened in case there were unexpected ones 

284 response.raise_for_status() 

285 

286 return response

288 def _send_request(self, request: requests.PreparedRequest, request_kwargs: Mapping[str, Any]) -> requests.Response: 

289 """ 

290 Creates backoff wrappers which are responsible for retry logic 

291 """ 

292 

293 """ 

294 Backoff package has max_tries parameter that means total number of 

295 tries before giving up, so if this number is 0 no calls expected to be done. 

296 But for this class we call it max_REtries assuming there would be at 

297 least one attempt and some retry attempts, to comply this logic we add 

298 1 to expected retries attempts. 

299 """ 

300 max_tries = self.max_retries 

301 """ 

302 According to backoff max_tries docstring: 

303 max_tries: The maximum number of attempts to make before giving 

304 up ...The default value of None means there is no limit to 

305 the number of tries. 

306 This implies that if max_tries is excplicitly set to None there is no 

307 limit to retry attempts, otherwise it is limited number of tries. But 

308 this is not true for current version of backoff packages (1.8.0). Setting 

309 max_tries to 0 or negative number would result in endless retry atempts. 

310 Add this condition to avoid an endless loop if it hasnt been set 

311 explicitly (i.e. max_retries is not None). 

312 """ 

313 if max_tries is not None: 

314 max_tries = max(0, max_tries) + 1 

315 

316 user_backoff_handler = user_defined_backoff_handler(max_tries=max_tries)(self._send) 

317 backoff_handler = default_backoff_handler(max_tries=max_tries, factor=self.retry_factor) 

318 return backoff_handler(user_backoff_handler)(request, request_kwargs)

320 def read_records( 

321 self, 

322 sync_mode: SyncMode, 

323 cursor_field: List[str] = None, 

324 stream_slice: Mapping[str, Any] = None, 

325 stream_state: Mapping[str, Any] = None, 

326 ) -> Iterable[Mapping[str, Any]]: 

327 stream_state = stream_state or {} 

328 pagination_complete = False 

329 

330 next_page_token = None 

331 while not pagination_complete: 

332 request_headers = self.request_headers(stream_state=stream_state, stream_slice=stream_slice, next_page_token=next_page_token) 

333 request = self._create_prepared_request( 

334 path=self.path(stream_state=stream_state, stream_slice=stream_slice, next_page_token=next_page_token), 

335 headers=dict(request_headers, **self.authenticator.get_auth_header()), 

336 params=self.request_params(stream_state=stream_state, stream_slice=stream_slice, next_page_token=next_page_token), 

337 json=self.request_body_json(stream_state=stream_state, stream_slice=stream_slice, next_page_token=next_page_token), 

338 data=self.request_body_data(stream_state=stream_state, stream_slice=stream_slice, next_page_token=next_page_token), 

339 ) 

340 request_kwargs = self.request_kwargs(stream_state=stream_state, stream_slice=stream_slice, next_page_token=next_page_token) 

341 

342 if self.use_cache: 

343 # use context manager to handle and store cassette metadata 

344 with self.cache_file as cass: 

345 self.cassete = cass 

346 # vcr tries to find records based on the request, if such records exist, return from cache file 

347 # else make a request and save record in cache file 

348 response = self._send_request(request, request_kwargs) 

349 

350 else: 

351 response = self._send_request(request, request_kwargs) 

352 

353 yield from self.parse_response(response, stream_state=stream_state, stream_slice=stream_slice) 

354 

355 next_page_token = self.next_page_token(response) 

356 if not next_page_token: 

357 pagination_complete = True 

358 

359 # Always return an empty generator just in case no records were ever yielded 

360 yield from []

371 def stream_slices( 

372 self, sync_mode: SyncMode, cursor_field: List[str] = None, stream_state: Mapping[str, Any] = None 

373 ) -> Iterable[Optional[Mapping[str, Any]]]: 

374 parent_stream_slices = self.parent.stream_slices( 

375 sync_mode=SyncMode.full_refresh, cursor_field=cursor_field, stream_state=stream_state 

376 ) 

377 

378 # iterate over all parent stream_slices 

379 for stream_slice in parent_stream_slices: 

380 parent_records = self.parent.read_records( 

381 sync_mode=SyncMode.full_refresh, cursor_field=cursor_field, stream_slice=stream_slice, stream_state=stream_state 

382 ) 

383 

384 # iterate over all parent records with current stream_slice 

385 for record in parent_records: 

386 yield {"parent": record}
